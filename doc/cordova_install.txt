# install ant
sudo apt-get install ant

# download, install, and symlink android SDK
https://developer.android.com/sdk/installing/index.html
#symlink android-sdk-linux/platform-tools/adb and android-sdk-linux/tools/android to your bin

# create 'hello world' app
# Details here: http://cordova.apache.org/docs/en/3.0.0/guide_cli_index.md.html
cd ~/src # or whatever you want as the parent dir
cordova create hello com.example.hello "Hello World"
cd hello
cordova platform add android
cordova build
cordova build android

# To run on an android emulator
cordova emulate android

# To run on a device:
cordova run android
